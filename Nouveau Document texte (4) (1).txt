local UserInputService = game:GetService("UserInputService")
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")

local LocalPlayer = Players.LocalPlayer
local camera = workspace.CurrentCamera

local connection
local lockedTarget = nil -- store locked player

-- GUI
local gui = Instance.new("ScreenGui")
local textLabel = Instance.new("TextLabel")

gui.Name = "CursorLockGui"
gui.Parent = LocalPlayer:WaitForChild("PlayerGui")

textLabel.Parent = gui
textLabel.Size = UDim2.new(0, 200, 0, 50)
textLabel.Position = UDim2.new(0.5, -100, 0, 0)
textLabel.TextScaled = true
textLabel.BackgroundTransparency = 0.5
textLabel.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
textLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
textLabel.Text = "Cursor Unlocked"

local Teams = {
	CounterForces = {"Class-D", "Chaos Insurgency"},
	Foundation = {"Security Department", "Mobile Task Forces", "Foundation Personnel"},
	Anomalies = {"SCP", "Serpents Hand"},
	UNGOG = {"Global Occult Coalition"},
}

local function Allies(T1, T2)
	for _, TeamList in pairs(Teams) do
		local FoundT1 = false
		local FoundT2 = false

		for _, TeamName in ipairs(TeamList) do
			if T1 == TeamName then FoundT1 = true end
			if T2 == TeamName then FoundT2 = true end
		end

		if FoundT1 and FoundT2 then
			return true
		end
	end

	return false
end

local function getClosestToMouse()
	local closestHead = nil
	local shortestDistance = math.huge

	for _, player in pairs(Players:GetPlayers()) do
		if player ~= LocalPlayer and not Allies(LocalPlayer.Team.Name, player.Team.Name) and player.Team ~= LocalPlayer.Team and player.Team.Name ~= "Lobby" and player.Character and player.Character:FindFirstChild("Head") then
			local head = player.Character.Head
			local screenPos, onScreen = camera:WorldToViewportPoint(head.Position)

			if onScreen then
				local mousePos = UserInputService:GetMouseLocation()
				local distance = (Vector2.new(screenPos.X, screenPos.Y) - mousePos).Magnitude

				if distance < shortestDistance then
					shortestDistance = distance
					closestHead = head
				end
			end
		end
	end

	return closestHead
end

local function startLock()
	lockedTarget = getClosestToMouse()

	if not lockedTarget then return end
	if LocalPlayer.Team.Name == "Lobby" then return end

	UserInputService.MouseBehavior = Enum.MouseBehavior.LockCenter
	textLabel.Text = "Cursor Locked"

	connection = RunService.RenderStepped:Connect(function()
		if lockedTarget and lockedTarget.Parent then
			camera.CFrame = CFrame.new(camera.CFrame.Position, lockedTarget.Position)
		else
			textLabel.Text = "Target Lost"
		end
	end)
end

local function stopLock()
	UserInputService.MouseBehavior = Enum.MouseBehavior.Default
	textLabel.Text = "Cursor Unlocked"

	if connection then
		connection:Disconnect()
		connection = nil
	end

	lockedTarget = nil
end

UserInputService.InputBegan:Connect(function(input, gameProcessed)
	if gameProcessed then return end

	if input.UserInputType == Enum.UserInputType.MouseButton2 then
		startLock()
	end
end)

UserInputService.InputEnded:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton2 then
		stopLock()
	end
end)
